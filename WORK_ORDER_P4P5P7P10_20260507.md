### 🥇 P4 — 단일 인스턴스 락 (Win32 Mutex)

**파일:** `main_webview.py` (현재 578줄)
**목적:** exe를 두 번 클릭해도 두 번째 인스턴스가 즉시 종료 → DB 충돌 방지

**변경 명세:**
```python
# main_webview.py 상단 import 블록에 추가
import ctypes

# main() 함수 시작 직후 첫 번째 코드로 삽입
_MUTEX_NAME = "SQM_Inventory_SingleInstance_Mutex"

def _acquire_single_instance_lock():
    mutex = ctypes.windll.kernel32.CreateMutexW(None, True, _MUTEX_NAME)
    if ctypes.windll.kernel32.GetLastError() == 183:  # ERROR_ALREADY_EXISTS
        ctypes.windll.kernel32.CloseHandle(mutex)
        return False
    return True

# main() 함수 맨 처음:
if not _acquire_single_instance_lock():
    import sys
    sys.exit(0)  # 두 번째 인스턴스 즉시 종료
```

**주의:**
- `ctypes`는 Python 표준 라이브러리 → 추가 설치 없음
- Windows 전용 (`windll`) — Mac/Linux 무시 (이 프로그램은 Windows Only)
- `sys.exit(0)` → 조용히 종료 (에러 메시지 없음)
- **Rule 5 적용: 300줄↑ → Python 스크립트로 수정**

**수정 스크립트:**
```python

'''
content = content.replace(

# 3) main() 함수 첫 줄에 락 획득 코드 삽입
```

### 🥈 P5 — JS Heartbeat (백엔드 silent fail 감지)

**파일:** `frontend/js/sqm-core.js` (신규 코드 추가)
**목적:** FastAPI 백엔드가 중간에 꺼지면 15초 이내에 사용자에게 배너 표시

**변경 명세 (sqm-core.js 하단에 추가):**
```javascript
// === P5: Backend Heartbeat ===
(function() {
  if (window.__SQM_HEARTBEAT_INSTALLED__) return;
  window.__SQM_HEARTBEAT_INSTALLED__ = true;

  var HEALTH_URL = (window.ENDPOINTS && window.ENDPOINTS.health)
    ? window.ENDPOINTS.health
    : '/api/health';
  var INTERVAL_MS = 15000;   // 15초마다 체크
  var FAIL_COUNT   = 0;
  var MAX_FAIL     = 2;       // 2회 연속 실패 시 배너 표시
  var _banner      = null;

  function showOfflineBanner() {
    if (_banner) return;
    _banner = document.createElement('div');
    _banner.id = 'sqm-offline-banner';
    _banner.style.cssText = [
      'position:fixed', 'top:0', 'left:0', 'right:0',
      'background:#c0392b', 'color:#fff',
      'text-align:center', 'padding:8px',
      'font-size:14px', 'z-index:99999',
      'font-family:Malgun Gothic,sans-serif'
    ].join(';');
    _banner.textContent = '⚠️ 서버 연결이 끊겼습니다. 프로그램을 재시작해 주세요.';
    document.body.appendChild(_banner);
  }

  function hideOfflineBanner() {
    if (_banner) {
      _banner.remove();
      _banner = null;
    }
  }

  function checkHealth() {
    fetch(HEALTH_URL, { method: 'GET', cache: 'no-store' })
      .then(function(r) {
        if (r.ok) {
          FAIL_COUNT = 0;
          hideOfflineBanner();
        } else {
          FAIL_COUNT++;
          if (FAIL_COUNT >= MAX_FAIL) showOfflineBanner();
        }
      })
      .catch(function() {
        FAIL_COUNT++;
        if (FAIL_COUNT >= MAX_FAIL) showOfflineBanner();
      });
  }

  // 앱 로드 후 5초 뒤 첫 체크, 이후 15초 간격
  setTimeout(function() {
    checkHealth();
    setInterval(checkHealth, INTERVAL_MS);
  }, 5000);
})();
```
